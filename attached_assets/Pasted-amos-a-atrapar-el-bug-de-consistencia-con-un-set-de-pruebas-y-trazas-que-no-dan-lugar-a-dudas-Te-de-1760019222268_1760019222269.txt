2) Watchdog de consistencia (frontend, DEV only)
Compara la misma fuente de verdad (selectores) entre lista y detalle para el mismo projectId. Si difieren, loggea un diff claro.
// consistencyWatchdog.ts
type VM = {
  costDisplay?: number;
  revenueDisplay?: number;
  currencyNative?: 'ARS'|'USD';
  budgetUtilization?: number;
  efficiency?: number;
  _v?: Record<string, number>;  // versionado por slice
};

export function assertConsistency({
  source, // 'list' | 'detail'
  projectId,
  listVM,
  detailVM
}: {
  source: 'list' | 'detail';
  projectId: string | number;
  listVM?: VM;
  detailVM?: VM;
}) {
  if (!listVM || !detailVM) return;
  const keys: (keyof VM)[] = [
    'costDisplay','revenueDisplay','currencyNative','budgetUtilization','efficiency'
  ];
  const diffs: Record<string,{list:any;detail:any}> = {};
  for (const k of keys) {
    if (listVM[k] !== detailVM[k]) {
      diffs[k as string] = { list: listVM[k], detail: detailVM[k] };
    }
  }
  if (Object.keys(diffs).length) {
    // ⚠️ Señal clara y accionable
    // eslint-disable-next-line no-console
    console.groupCollapsed(`[CONSISTENCY] project ${projectId} mismatch (${source})`);
    console.table(diffs);
    console.log('list._v', listVM._v, 'detail._v', detailVM._v);
    console.groupEnd();
  }
}

3) Timeline de estado + checksum (detecta “parches que borran”)
Loggea cada aplicación de patch con un hash barato por slice. Si summary desaparece entre ticks, se reporta.
// stateTimeline.ts
const hash = (o:any) => {
  try { return btoa(unescape(encodeURIComponent(JSON.stringify(o))).slice(0,48)); }
  catch { return 'NA'; }
};

export function logPatch(prev:any, next:any, tag:string){
  const fields = ['summary','quotation','actuals','metrics'];
  const row:any = { tag, ts: new Date().toISOString() };
  for (const f of fields) {
    row[`${f}_prev`] = prev?.[f] ? hash(prev[f]) : '-';
    row[`${f}_next`] = next?.[f] ? hash(next[f]) : '-';
  }
  // eslint-disable-next-line no-console
  console.table(row);

  // Invariante fuerte: nunca “bajar” summary a undefined por un patch parcial
  if (prev?.summary && !next?.summary) {
    // eslint-disable-next-line no-console
    console.error('[INVARIANT] summary dropped by patch:', tag);
  }
}
