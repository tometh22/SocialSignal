2) Reglas de negocio (SoT → ViewModel)
2.1. Team (por persona)
De fact_labor_month (filtrado por project_id, period_key):
name (join dim_person)
roleName
targetHours ← target_hours
hoursAsana ← asana_hours
hoursBilling ← billing_hours (¡usar esta para costo!)
hourlyRateARS← hourly_rate_ars
fx ← fx
costARS ← cost_ars (o billing_hours * hourly_rate_ars)
costUSD ← cost_usd
flags
2.2. Totales (operativa)
totalAsanaHours = Σ asana_hours
totalBillingHours = Σ billing_hours
totalCostARS = Σ cost_ars
totalCostUSD = Σ cost_usd
2.3. Ingresos y cotización (Rendimiento Cliente)
De fact_rc_month (mismo filtro):
revenue_ars, revenue_usd, cost_ars, cost_usd, price_native, currency_native
currencyNative = currency_native del mes.
denominador (para presupuesto y métricas) = price_native (lo que en tu hoja llamás Cotización/Precio nativo del mes).
2.4. Vistas y métricas
Vista Operativa
Si currencyNative='ARS':
costDisplay = totalCostARS
revenueDisplay = revenue_ars
Si currencyNative='USD':
costDisplay = totalCostUSD
revenueDisplay = revenue_usd
markup = revenueDisplay / costDisplay (si ambos > 0)
margin = (revenueDisplay - costDisplay) / revenueDisplay
budgetUtilization (lo que muestra “% Presupuesto usado” del Dashboard)
= costDisplay / quotation.totalAmountNative
y quotation.totalAmountNative = price_native de Rendimiento Cliente (tu regla).
Nota: esto replica exactamente lo que validaste manualmente para KC Agosto (8,450 / 3,276,539, etc.) en la moneda nativa del proyecto.

// pseudo-implementación, orientativa
import { db } from '../db';

export async function getProjectPeriodViewFromSoT(projectId: number, periodKey: string) {
  // 1) Team breakdown
  const teamRows = await db.query(`
    SELECT p.name, flm.role_name, flm.target_hours, flm.asana_hours, flm.billing_hours,
           flm.hourly_rate_ars, flm.fx, flm.cost_ars, flm.cost_usd, flm.flags
    FROM fact_labor_month flm
    JOIN dim_person p ON p.id = flm.person_id
    WHERE flm.project_id = $1 AND flm.period_key = $2
    ORDER BY p.name
  `, [projectId, periodKey]);

  // 2) Totales SoT
  const totals = await db.query(`
    SELECT
      COALESCE(SUM(target_hours),0)  AS target_hours,
      COALESCE(SUM(asana_hours),0)   AS asana_hours,
      COALESCE(SUM(billing_hours),0) AS billing_hours,
      COALESCE(SUM(cost_ars),0)      AS cost_ars,
      COALESCE(SUM(cost_usd),0)      AS cost_usd
    FROM fact_labor_month
    WHERE project_id = $1 AND period_key = $2
  `, [projectId, periodKey]);

  // 3) Rendimiento Cliente del mes
  const rc = await db.query(`
    SELECT currency_native, price_native,
           COALESCE(SUM(revenue_ars),0) AS revenue_ars,
           COALESCE(SUM(revenue_usd),0) AS revenue_usd,
           COALESCE(SUM(cost_ars),0)    AS rc_cost_ars,
           COALESCE(SUM(cost_usd),0)    AS rc_cost_usd
    FROM fact_rc_month
    WHERE project_id = $1 AND period_key = $2
    GROUP BY currency_native, price_native
    LIMIT 1
  `, [projectId, periodKey]);

  if (!rc.rows.length && !teamRows.rows.length) return null; // sin datos SoT

  const t = totals.rows[0] || {};
  const r = rc.rows[0] || { currency_native: 'ARS', price_native: 0, revenue_ars: 0, revenue_usd: 0 };

  // 4) Vista operativa en moneda nativa
  const currencyNative = r.currency_native || 'ARS';
  const costDisplay    = currencyNative === 'USD' ? Number(t.cost_usd) : Number(t.cost_ars);
  const revenueDisplay = currencyNative === 'USD' ? Number(r.revenue_usd) : Number(r.revenue_ars);

  const summary = {
    currencyNative,
    costDisplay,
    revenueDisplay,
    markup: (costDisplay > 0 && revenueDisplay > 0) ? (revenueDisplay / costDisplay) : null,
    margin: (revenueDisplay > 0) ? ((revenueDisplay - costDisplay) / revenueDisplay) : null,
  };

  const quotation = {
    totalAmountNative: Number(r.price_native || 0)  // tu “precio nativo / cotización” del mes
  };

  const actuals = {
    totalAsanaHours:   Number(t.asana_hours || 0),
    totalBillingHours: Number(t.billing_hours || 0),
    teamBreakdown: teamRows.rows.map(x => ({
      name: x.name,
      roleName: x.role_name,
      targetHours:  Number(x.target_hours || 0),
      hoursAsana:   Number(x.asana_hours || 0),
      hoursBilling: Number(x.billing_hours || 0),
      hourlyRateARS: Number(x.hourly_rate_ars || 0),
      fx: Number(x.fx || 0),
      costARS: Number(x.cost_ars || 0),
      costUSD: Number(x.cost_usd || 0),
      flags: x.flags,
    }))
  };

  return { summary, quotation, actuals };
}
