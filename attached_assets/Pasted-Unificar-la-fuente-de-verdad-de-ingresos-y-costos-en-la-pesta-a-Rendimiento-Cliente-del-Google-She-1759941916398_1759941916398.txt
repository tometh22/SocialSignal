2) Agregador único para API

Modificar/crear server/aggregators/projects.ts para leer solo de financial_sot:

// currency native
const isUSD = (client:string)=> /warner|kimberly/i.test(client);
const toARS = (usd:number, fx:number)=> (usd ?? 0) * (fx ?? 0);

// para cada fila financial_sot:
const revenueUSD = row.revenue_usd ?? 0;
const costUSD    = row.cost_usd ?? 0;
const fx         = row.fx ?? fxTable[period];

const currencyNative = isUSD(row.client_name) ? "USD" : "ARS";
const revenueDisplay = currencyNative === "USD" ? revenueUSD : toARS(revenueUSD, fx);
const costDisplay    = currencyNative === "USD" ? costUSD    : toARS(costUSD, fx);

const markup = costUSD > 0 ? revenueUSD / costUSD : null;
const margin = revenueUSD > 0 ? (revenueUSD - costUSD) / revenueUSD : null;

return {
  clientName: row.client_name,
  projectName: row.project_name,
  tags: [row.project_type].filter(Boolean),
  currencyNative,
  metrics: {
    revenueDisplay, costDisplay,                // nativo (ARS o USD)
    revenueUSDNormalized: revenueUSD,          // KPIs en USD
    costUSDNormalized: costUSD,
    markup, margin
  }
};


Resumen del período = sumas en USD (revenue_usd, revenue_usd - cost_usd).
