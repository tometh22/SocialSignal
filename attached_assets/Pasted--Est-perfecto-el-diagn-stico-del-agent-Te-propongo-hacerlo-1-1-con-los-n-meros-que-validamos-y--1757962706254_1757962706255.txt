// util: parsear decimales con coma o puntos
const parseDec = (v: unknown) =>
  typeof v === "number" ? v :
  Number(String(v).replace(/\./g, "").replace(",", ".")) || 0;

// --- Eficiencia
const L = parseDec(row[columnMap.horasRealesAsana]);   // reales
const K = parseDec(row[columnMap.horasObjetivo]);      // objetivo

const effRatio = K === 0 ? null : Math.min(1, L / K);
const effScore = K === 0 ? 70 : (effRatio! * 100);     // << 70 pts neutro

// redondeo SOLO para mostrar
const effScoreUi = Math.round(effScore * 100) / 100;
2) Impacto: participación * eficiencia_normalizada * 30
Participación del período/proyecto-mes = peso de cada fila / suma del grupo.
Peso = M (horas facturadas) × ValorHora. (Si tenés montoUSD, podés usarlo; para participación da igual la moneda porque se cancela).
// Clave de agrupación recomendada (proyecto + YYYY-MM)
const periodo = `${row.year}-${String(row.month).padStart(2,"0")}-01`;
const groupKey = `${row.proyectoId}|${periodo}`;

// Peso por fila (fallback si no hay USD)
const M  = parseDec(row[columnMap.horasFacturacion]);
const VH = parseDec(row[columnMap.valorHoraARS]);
const montoUSD = parseDec(row[columnMap.montoUSD]); // si existe
const pesoFila = Number.isFinite(montoUSD) && montoUSD > 0 ? montoUSD : (M * VH);

// Suma por grupo (hecha con reduce por groupKey)
const totalGrupo = sumBy(grupos[groupKey], "pesoFila");
const share = totalGrupo > 0 ? (pesoFila / totalGrupo) : 0;

// eficiencia normalizada para impacto (K=0 ⇒ 1 para no penalizar)
const effNorm = K === 0 ? 1 : Math.min(1, L / K);

// Impacto 0–30 (sin redondear)
const impacto = effNorm * share * 30;
const impactoUi = Math.round(impacto * 100) / 100;
3) Unificado: 50% eficiencia + 50% impacto (escalado)
const impactoEsc = (impacto / 30) * 100; // 0–100
const unificado = 0.5 * effScore + 0.5 * impactoEsc;
const unificadoUi = Math.round(unificado * 100) / 100;
