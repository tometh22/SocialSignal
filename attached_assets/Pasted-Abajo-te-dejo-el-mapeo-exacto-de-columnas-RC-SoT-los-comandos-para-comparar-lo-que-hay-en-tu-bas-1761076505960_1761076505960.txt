2.2. Cruce Labor vs RC (invariante: Δ≈0)
$PSQL <<'SQL'
WITH labor AS (
  SELECT period_key, 
         ROUND(SUM(cost_usd)::numeric,2) AS labor_cost_usd
  FROM fact_labor_month
  WHERE project_id = :PROJECT_ID
  GROUP BY period_key
),
rc AS (
  SELECT period_key, ROUND(cost_usd::numeric,2) AS rc_cost_usd
  FROM fact_rc_month
  WHERE project_id = :PROJECT_ID
)
SELECT l.period_key,
       l.labor_cost_usd,
       r.rc_cost_usd,
       ROUND((l.labor_cost_usd - r.rc_cost_usd)::numeric,2) AS delta_usd
FROM labor l JOIN rc r USING(period_key)
ORDER BY l.period_key;
SQL
Esperado: delta_usd ∈ {-0.01, 0.00, 0.01}.
