app.get('/internal/rebuild-sot', async (req, res) => {
  const period = String(req.query.period);
  const ledger = await getPeriodLedger(period, { fresh: true }); // mismo módulo de la lista
  // volcar a SOT

    await t.none('DELETE FROM costs_sot  WHERE month_key=$1', [period]);
    for (const r of ledger) {
      await t.none(`
        INSERT INTO income_sot (project_key, month_key, currency_native, revenue_display, revenue_usd, flags)
        VALUES ($1,$2,$3,$4,$5,$6)
      `, [r.projectKey, period, r.currencyNative, r.revenueDisplay, r.revenueUSD, r.flags]);
      await t.none(`
        INSERT INTO costs_sot (project_key, month_key, currency_native, cost_display, cost_usd, flags)
        VALUES ($1,$2,$3,$4,$5,$6)
      `, [r.projectKey, period, r.currencyNative, r.costDisplay, r.costUSD, r.flags]);
    }
  });
  res.json({ ok: true, period, rows: ledger.length });
});

2) Fix robusto del ETL (septiembre en adelante)
2.1. Sanitización dura en ingestión

Anti-USD astronómico: si monto_total_usd > 100000 ⇒ ignorar USD y convertir desde ARS/FX.

Anti-×100 horas: si horas_reales_asana > 300 ⇒ dividir costo_total por 100 aunque valor_hora_persona sea 0.

Decimal parser unificado: parseNumberEs que respete 2277.78 y 2.277,78.

Ejemplo (pseudo-TS en tu ETL):
const FX = await getMonthlyFx(period);

function sanitizeCostRow(r){
  const flags:string[] = [];

  let ars = r.costo_total_ars ?? 0;
  let usd = r.monto_total_usd ?? null;

  if (usd != null && usd > 100000) { // USD astronómico
    flags.push('USD_CORRUPTED_IGNORED');
    usd = null;
  }

  if (r.horas_reales_asana > 300) {
    flags.push('ANTI_X100_HOURS');
    ars = ars / 100;
  }

  if (usd == null) {
    usd = ars > 0 ? ars / FX : 0;
  }

  return { ars, usd, flags };
}
