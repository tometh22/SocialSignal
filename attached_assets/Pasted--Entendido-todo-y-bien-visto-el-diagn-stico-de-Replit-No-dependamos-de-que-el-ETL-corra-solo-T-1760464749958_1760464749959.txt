// complete-data.ts (al mapear actuals.teamBreakdown)
function normHours(x?: number) {
  if (!Number.isFinite(x as number)) return 0;
  return (x as number) > 500 ? (x as number) / 100 : (x as number);
}

function hydrateMember(m:any, fxMes:number) {
  const targetHours   = Number(m.targetHours ?? m.estimatedHours ?? 0); // vienen bien
  const hoursRaw      = Number(m.hours ?? m.actualHours ?? 0);          // legacy “inflado”
  const asanaFixed    = Number.isFinite(m.hoursAsana) ? m.hoursAsana : normHours(hoursRaw);

  const billingRaw    = Number(m.hoursBilling ?? m.hoursParaFacturacion ?? 0);
  const hoursBilling  = billingRaw ? normHours(billingRaw)
                        : asanaFixed ? asanaFixed
                        : targetHours;

  const rateARS       = Number(m.hourlyRateARS ?? m.rateARS ?? 0);
  const costARS       = Number(m.costARS ?? (hoursBilling * rateARS) || 0);
  const costUSD       = Number(m.costUSD ?? (fxMes ? costARS / fxMes : 0));

  return {
    name: m.name, roleName: m.roleName ?? 'N/A',
    targetHours,
    hoursAsana: asanaFixed,
    hoursBilling,
    hourlyRateARS: rateARS,
    costARS,
    costUSD,
    // compat
    hours: asanaFixed,
  };
}

const fxMes = payload.fx || rc.fx_mes; // según tu contrato
const tb    = (dbTeamBreakdown || []).map(m => hydrateMember(m, fxMes));

const totalAsanaHours   = tb.reduce((a,b)=>a+(b.hoursAsana||0),0);
const totalBillingHours = tb.reduce((a,b)=>a+(b.hoursBilling||0),0);
