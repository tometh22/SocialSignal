Ejemplo ultra-simple del guard (TypeScript):

app.get('/api/projects/:client/:project/costs', async (req, res) => {
  const { client, project } = req.params;
  const { period } = req.query as { period?: string };
  if (!client || !project || !period) {
    return res.status(400).json({ error: 'client, project y period son requeridos' });
  }
  const data = await getCostsForProject({ clientName: client, projectName: project, period });
  res.json(data);
});
