2) Asegurá la lectura correcta de montos
// Helper robusto
const toNumber = (v: unknown): number | null => {
  if (v === null || v === undefined) return null;
  const s = String(v).replace(/[\s$,]/g, "").replace(",", "."); // UNFORMATTED_VALUE igual viene limpio, esto es por las dudas
  const n = Number(s);
  return Number.isFinite(n) ? n : null;
};

// Map de columnas
const montoUsd = toNumber(r.montoTotalUsd);  // Columna R
const montoArs = toNumber(r.monedasArs);     // Columna O (si la usás de respaldo)
const fx       = toNumber(r.cotizacion);     // Columna Q

// Regla: priorizar USD de la col. R; si está vacío, derivar de ARS/fx
const amountUsd = montoUsd ?? (fx && montoArs ? montoArs / fx : null);

// Ignorar filas sin importe (>0) o sin período válido
if (!amountUsd || amountUsd <= 0) return false;
if (!periodKey /* YYYY-MM derivado de columnas C (Mes) + D (Año) */) return false;

3) Insert/Upsert en fact_cost_month
await db
