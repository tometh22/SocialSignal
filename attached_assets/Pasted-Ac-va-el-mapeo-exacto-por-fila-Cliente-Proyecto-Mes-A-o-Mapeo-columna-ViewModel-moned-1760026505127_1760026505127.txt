Pseudocódigo del agregador (backend)
const USD_NATIVE = new Set(['Warner', 'Kimberly Clark']);
const cur = USD_NATIVE.has(cliente) ? 'USD' : 'ARS';

const costUSD = row['Costos (USD)'] ?? 0;
const billUSD = row['Facturación (USD)'] ?? 0;
const quote   = row['Cotización'] ?? 0; // ya en nativa (convención hoja)

const fx = rates.usd_ars; // del día/mes

const costDisplay = cur === 'USD' ? costUSD : Math.round(costUSD * fx);
const revenueDisplay = cur === 'USD' ? billUSD : Math.round(billUSD * fx);

return {
  summary: { costDisplay, revenueDisplay, currencyNative: cur, markup: quote && costDisplay ? quote/costDisplay : null,
             margin: quote ? (quote - costDisplay)/quote : null },
  quotation: { totalAmount: quote },
  actuals: { totalWorkedCost: costUSD } // USD base
};
