2) Baseline robusto (ignorar 0/near-zero)

En buildBaselineUSD(...):

function buildBaselineUSD(history: number[], minUSD=0): number | null {
  const values = history.filter(x => Number.isFinite(x) && x >= minUSD);
  if (values.length < 1) return null;
  values.sort((a,b)=>a-b);
  const mid = Math.floor(values.length/2);
  return values.length % 2 ? values[mid] : (values[mid-1] + values[mid]) / 2;
}


Y al llamar:

const baselineUSD = buildBaselineUSD(historyUSDNormalized, cfg.min_baseline_usd ?? 0);

3) Preferir prev_valid y clamp de autocorrección

Dentro del bloque donde aplicás el fix:

const prevValid = historyUSDNormalized.find(x => Number.isFinite(x) && x >= (cfg.min_baseline_usd ?? 0)) ?? null;

if (decision.isAnomaly && decision.fixedUSD != null) {
  // Clamp contra el mes válido anterior si existe
  if (prevValid) {
    const lo = prevValid * (cfg.lower_clamp_ratio ?? 0.6);
    const hi = prevValid * (cfg.upper_clamp_ratio ?? 1.4);
    decision.fixedUSD = Math.min(hi, Math.max(lo, decision.fixedUSD));
    outFlags.push('CLAMPED_TO_PREV_RANGE');
  }
  usdNormalized = decision.fixedUSD;
  displayNative = (nativeCurrency === 'ARS') ? Math.round(usdNormalized * fx) : usdNormalized;
}
