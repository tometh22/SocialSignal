Toque mínimo en TCG (pseudocódigo):

const prevValid = getPrevValidUSD(..., lookback_months, cfg.min_baseline_usd);
const nextValid = cfg.use_future_baseline ? getNextValidUSD(..., cfg.lookahead_months, cfg.min_baseline_usd) : null;

let fixedUSD = prevValid ?? nextValid ?? baselineMedian;

if (cfg.expected_usd_band) {
  const {low, high} = cfg.expected_usd_band;
  fixedUSD = Math.min(high, Math.max(low, fixedUSD));   // clamp a banda esperada
  flags.push('CLAMPED_TO_EXPECTED_BAND');
} else if (prevValid) {
  // clamp suave alrededor del prevValid
  const lo = prevValid * (cfg.lower_clamp_ratio ?? 0.6);
  const hi = prevValid * (cfg.upper_clamp_ratio ?? 1.4);
  fixedUSD = Math.min(hi, Math.max(lo, fixedUSD));
  flags.push('CLAMPED_TO_PREV_RANGE');
}
