app.get('/api/projects', async (req, res) => {
  const timeFilter = String(req.query.timeFilter ?? 'this_month');
  const basis = String(req.query.basis ?? 'ECON');

  const rows = await buildProjectList({ timeFilter, basis }); // usa el motor único

  // Agrega un resumen opcional para los KPIs superiores
  const pageSummary = rows.reduce(
    (acc: any, p: any) => {
      const rev = Number(p.summary?.revenueUSD ?? 0);
      const hrs = Number(p.summary?.totalHours ?? 0);
      acc.totalProjects += 1;
      acc.active += (p.status === 'active') ? 1 : 0;
      acc.totalRevenueUSD += rev;
      acc.totalHours += hrs;
      return acc;
    },
    { totalProjects: 0, active: 0, totalRevenueUSD: 0, totalHours: 0 }
  );

  res.json({ summary: pageSummary, projects: rows });
});

async function buildProjectList({ timeFilter, basis }: { timeFilter: string; basis: 'ECON' | 'EXEC' | string; }) {
  const projects = await storage.getProjects(); // lo que uses
  const out = [];

  for (const p of projects) {
    const m = await computeProjectPeriodMetrics({ projectId: p.id, timeFilter, basis: 'ECON' }); // motor único

    // fila compatible con la UI actual
    out.push({
      projectId: p.id,
      clientName: p.client.name,
      projectName: p.quotation?.projectName ?? p.name,
      status: p.status,                         // "active" | "paused" | "completed"
      projectType: p.quotation?.projectType,    // "recurrente" | "one-time"

      // BLOQUE UNIVERSAL (siempre)
      summary: {
        period: m.periodKey,                      // "2025-08"
        basis: 'ECON',
        revenueUSD: Number(m.revenueUSD),         // 8450

        totalHours: Number(m.totalHours),         // 288.28
