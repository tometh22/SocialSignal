2) Data Provider interno (fallback)
function useTeamData({ projectId, periodKey, injected }: {
  projectId: string; periodKey: string;
  injected?: { teamBreakdown?: TeamMember[]; totalHours?: number; estimatedHours?: number; }
}) {
  const shouldFetch = !injected?.teamBreakdown?.length;
  const { data, isLoading, error } = useQuery(
    ['tda', projectId, periodKey],
    () => api.get(`/projects/${projectId}/team-deviation?period=${periodKey}`),
    { enabled: shouldFetch }
  );

  const tb = injected?.teamBreakdown ?? data?.teamBreakdown ?? [];
  const total = injected?.totalHours ?? data?.totalHours ?? tb.reduce((a,b)=>a+(b.hours||0),0);
  const est   = injected?.estimatedHours ?? data?.estimatedHours ?? tb.reduce((a,b)=>a+(b.targetHours||0),0);

  return { teamBreakdown: tb, totalHours: total, estimatedHours: est, isLoading: shouldFetch && isLoading, error };
}

Σtb.targetHours === estimatedHours
tb.every(x => !!x.roleName)
