Resolver universal:
// services/project-resolver.ts
import projectsByKey from "../config/projects.json";
import idMap from "../config/project-id-map.json";

export type ProjectConfig = typeof projectsByKey[keyof typeof projectsByKey];

export function resolveProject(input: { projectId?: string|number; projectKey?: string }): ProjectConfig {
  const key = input.projectKey ?? (input.projectId != null ? idMap[String(input.projectId)] : undefined);
  if (!key) throw new Error(`Unknown project (id=${input.projectId}, key=${input.projectKey})`);
  const cfg = (projectsByKey as any)[key];
  if (!cfg) throw new Error(`Missing config for key=${key}`);
  return cfg;
}

// server/routes.ts
import { resolveProject } from "../services/project-resolver";
import { computeRankings } from "../services/universal-rankings";
import { resolvePeriod } from "../services/period";

app.get("/api/projects/:projectId/performance-rankings", async (req, res, next) => {
  try {
    const { projectId } = req.params;
    const { timeFilter, start, end } = req.query as any;

    const cfg = resolveProject({ projectId });  // ← sin hardcode
    const period = resolvePeriod({ timeFilter, start, end }); // yyyy-mm ... yyyy-mm

    const rows = await loadRowsFromSheet(cfg);  // usa cfg.spreadsheetId + sheetName
    const filtered = filterByPeriod(rows, period);
    const rankings = computeRankings(filtered, cfg.columnMap);

    // compatibilidad: misma forma que el endpoint original
    res.set("X-Rankings-Engine", "universal");
    res.json({ rankings });
  } catch (e) { next(e); }

// scripts/verify-rankings.ts
import { resolveProject } from "../services/project-resolver";
import { computeRankings } from "../services/universal-rankings";
import { resolvePeriod, filterByPeriod } from "../services/period";
import exp from "./fixtures/expected-july-2025.json";

(async () => {
  const cfg = resolveProject({ projectId: 39 });
  const period = resolvePeriod({ timeFilter: "july_2025" });
  const rows = await loadRowsFromSheet(cfg);
  const results = computeRankings(filterByPeriod(rows, period), cfg.columnMap);

  const tol = 0.3;
  for (const [persona, expected] of Object.entries(exp as any)) {
    const r = results.find(x => x.person === persona);
    if (!r) throw new Error(`Missing ${persona}`);
    for (const k of ["unificado","eficiencia","impacto"] as const) {
      if (Math.abs((r as any)[k] - expected[k]) > tol) {
        throw new Error(`Mismatch ${persona}.${k}: got ${(r as any)[k]} expected ${expected[k]}`);
      }
    }
  }
  console.log("✅ Regression OK");
})();
