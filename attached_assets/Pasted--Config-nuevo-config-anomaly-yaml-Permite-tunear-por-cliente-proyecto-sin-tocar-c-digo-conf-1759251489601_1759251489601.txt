2) Guardas en ETL (evitar futuros USD corruptos)

No tocamos datos históricos aún; esto previene que vuelva a pasar.

// ETL (pseudocódigo/TS) al construir el payload a insertar en direct_costs
function sanitizeUSD(input: { currency: 'ARS'|'USD', costoARS?: number, montoUSD?: number, fx?: number }) {
  const usd = input.montoUSD ?? null;
  const ars = input.costoARS ?? null;
  const fx  = input.fx ?? null;

  // 1) Prohibido que USD == ARS cuando la moneda nativa es ARS
  if (input.currency === 'ARS' && usd != null && ars != null) {
    const same = Math.abs(usd - ars) <= 0.01 * Math.max(1, ars);
    if (same) return null;
  }

  // 2) Cortar astronómicos o notaciones científicas
  if (usd != null && (usd > 1_000_000 || !Number.isFinite(usd))) return null;

  // 3) Si hay ARS y FX, el USD de origen no debe diferir >X% del ARS/FX
  if (usd != null && ars != null && fx != null) {
    const implied = ars / fx;
    const diffPct = Math.abs(usd - implied) / Math.max(1, implied) * 100;
    if (diffPct > 200) return null; // 200% por si FX sufre saltos grandes
  }

  return usd ?? null;
}


Importante: en ETL nunca calcules USD si el origen es ARS. Guardá USD solo si viene explícito y pasa los checks.

3) Guardas en SoT (Temporal Consistency Guard – TCG)

Corre después de anti-USD-astronómico y anti-×100-horas.
Si detecta desvío temporal, reemplaza el valor actual por un fallback (mediana o mes válido anterior) y flaggea.

3.1 Helper para baseline
// Query previa: trae últimos N meses válidos (sin flags graves) con costUSDNormalized
// SELECT month_key, cost_usd_normalized FROM costs_sot WHERE projectKey=? AND is_valid=true ORDER BY month_key DESC LIMIT 6

function buildBaselineUSD(history: number[]): number | null {
  const values = history.filter(x => Number.isFinite(x));
  if (values.length < 2) return null;
  // mediana robusta
  values.sort((a,b)=>a-b);
  const mid = Math.floor(values.length/2);
  return values.length % 2 ? values[mid] : (values[mid-1] + values[mid]) / 2;
}

3.2 Detector + corrección
type AnomalyDecision = {
  isAnomaly: boolean;
  ratio?: number;
  baselineUSD?: number|null;
  fixedUSD?: number|null;
  reason?: string;
  flags: string[];
};

function temporalGuard({
  projectKey,
  monthKey,
  nativeCurrency,         // 'ARS' | 'USD'
  costNative,             // número “display” crudo tras saneamiento
  fx,                     // FX del mes
  montoUSDFromOrigin,     // USD limpio si vino de origen, sino null
  historyUSDNormalized,   // array con últimos meses válidos
  cfg                     // merge de defaults + override por projectKey
}: {
  projectKey: string; monthKey: string; nativeCurrency: 'ARS'|'USD';
  costNative: number; fx: number; montoUSDFromOrigin: number|null;
  historyUSDNormalized: number[]; cfg: AnomalyConfig;
}): AnomalyDecision {

  const flags: string[] = [];
  // 1) Elegir USD “actual” candidato
  const currentUSD = montoUSDFromOrigin ?? (nativeCurrency === 'ARS' ? costNative / fx : costNative);

  // 2) Baseline
  const baselineUSD = buildBaselineUSD(historyUSDNormalized);
  if (!baselineUSD) return { isAnomaly:false, flags };

  const ratio = currentUSD / baselineUSD;

  // 3) Detección
  const low = ratio < (cfg.low_ratio_threshold ?? 0.35);
  const high = ratio > (cfg.high_ratio_threshold ?? 2.5);
  if (!low && !high) return { isAnomaly:false, ratio, baselineUSD, flags };

  flags.push('ANOMALY_TEMPORAL');
  let fixedUSD: number|null = null;
  let reason = '';

  if (cfg.mode === 'autocorrect') {
    if ((cfg.strategy ?? 'median') === 'prev_valid' && historyUSDNormalized.length) {
      fixedUSD = historyUSDNormalized[0]; // último válido
      reason = 'AUTO_FROM_PREV_VALID';
    } else {
      fixedUSD = baselineUSD;             // mediana robusta
      reason = 'AUTO_FROM_MEDIAN';
    }
  }

  return { isAnomaly:true, ratio, baselineUSD, fixedUSD, reason, flags };
}

const outFlags = [...existingFlags, ...decision.flags];

if (decision.isAnomaly && decision.fixedUSD != null) {

  displayNative = (nativeCurrency === 'ARS') ? Math.round(usdNormalized * fx) : usdNormalized;
