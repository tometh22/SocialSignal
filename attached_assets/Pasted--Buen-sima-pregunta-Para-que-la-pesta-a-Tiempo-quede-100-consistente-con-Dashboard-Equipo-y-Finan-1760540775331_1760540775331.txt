router.get('/api/projects/:id/time-tracking', async (req, res) => {
  const { id } = req.params;
  const period = req.query.period as string; // '2025-08'
  const agg = await loadProjectAggregate(id, period); // mismo que complete-data

  const tb = (agg.actuals_data.teamBreakdown ?? []).map(hydrateLegacyHours(fxMes));
  const summary = {
    membersActive: tb.filter(m => (m.hoursAsana||0) > 0 || (m.hoursBilling||0) > 0).length,
    totalAsanaHours: sum(tb.map(m=>m.hoursAsana||0)),
    estimatedHours: sum(tb.map(m=>m.targetHours||0)),
  };
  const dias = businessDays(period);
  const avgDaily = summary.totalAsanaHours / (dias * Math.max(summary.membersActive,1));

  res.json({
    projectId: +id, period,
    summary: { ...summary, progressPct: safeDiv(summary.totalAsanaHours, summary.estimatedHours), avgDailyHoursPerMember: +avgDaily.toFixed(2) },
    members: tb.map(m => ({
      personId: m.personId, name: m.name, roleName: m.roleName,
      targetHours: m.targetHours, hoursAsana: m.hoursAsana, hoursBilling: m.hoursBilling,
      status: m.hoursAsana > m.targetHours ? 'exceso' : (m.hoursAsana === m.targetHours ? 'cumplido' : 'pendiente'),
      badges: m.hoursAsana > m.targetHours ? ['Completo','Exceso tiempo'] : (m.hoursAsana === m.targetHours ? ['Completo','Objetivo cumplido'] : [])
    }))
  });
});
