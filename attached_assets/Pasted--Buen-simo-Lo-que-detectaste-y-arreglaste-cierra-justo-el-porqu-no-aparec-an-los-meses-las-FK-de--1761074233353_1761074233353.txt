Smoke test de fin de obra
1) RC por período (debe existir y tener cotización + fx)
SELECT period_key,
       CAST(quote_native AS numeric(12,2)) AS cotizacion,
       CAST(fx_rate AS numeric(10,2))      AS fx,
       CAST(revenue_usd AS numeric(12,2))  AS ingreso_usd,
       CAST(cost_usd AS numeric(12,2))     AS costo_usd
FROM fact_rc_month
WHERE project_id = 39
ORDER BY period_key;
2) Labor por período (invariantes de horas y costos)
SELECT period_key,
       ROUND(SUM(target_hours)::numeric,2)  AS target_h,
       ROUND(SUM(asana_hours)::numeric,2)   AS asana_h,
       ROUND(SUM(billing_hours)::numeric,2) AS billing_h,
       ROUND(SUM(cost_usd)::numeric,2)      AS cost_usd
FROM fact_labor_month
WHERE project_id = 39
GROUP BY period_key
ORDER BY period_key;
3) Invariante Labor vs RC (Δ debe ser ~0)
WITH labor AS (
  SELECT period_key, SUM(cost_usd) cost_usd
  FROM fact_labor_month
  WHERE project_id = 39 GROUP BY period_key
),
rc AS (
  SELECT period_key, cost_usd FROM fact_rc_month
  WHERE project_id = 39
)
SELECT l.period_key,
       ROUND(l.cost_usd::numeric,2) AS labor_cost_usd,
       ROUND(r.cost_usd::numeric,2) AS rc_cost_usd,
       ROUND((l.cost_usd - r.cost_usd)::numeric,4) AS delta
FROM labor l JOIN rc r USING (period_key)
ORDER BY l.period_key;
