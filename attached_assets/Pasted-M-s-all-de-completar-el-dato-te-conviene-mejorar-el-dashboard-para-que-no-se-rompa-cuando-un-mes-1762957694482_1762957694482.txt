SQL/Drizzle (snippets clave)
Billed (USD)
const [{ billedUsd }] = await db
  .select({ billedUsd: sql<number>`coalesce(sum(${fr.revenueUsd}),0)`.mapWith(Number) })
  .from(fr).where(and(eq(fr.orgId, orgId), eq(fr.periodKey, periodKey)));
Costs (USD)
const [{ costUsd }] = await db
  .select({ costUsd: sql<number>`coalesce(sum(${fr.costUsd}),0)`.mapWith(Number) })
  .from(fr).where(and(eq(fr.orgId, orgId), eq(fr.periodKey, periodKey)));
FX ponderado
const [{ fxWeighted }] = await db
  .select({ fxWeighted: sql<number>`
    coalesce(
      sum(${fr.revenueArs} + ${fr.costArs})
      / nullif(sum(${fr.revenueUsd} + ${fr.costUsd}),0), 0
    )`.mapWith(Number) })
  .from(fr).where(and(eq(fr.orgId, orgId), eq(fr.periodKey, periodKey)));
Horas (billable / non-billable)
const [{ total, billable, nonBillable }] = await db
  .select({
    total:       sql<number>`coalesce(sum(${fl.asanaHours}),0)`.mapWith(Number),
    billable:    sql<number>`coalesce(sum(${fl.asanaHours}) filter (where ${fl.isBillable}=true),0)`.mapWith(Number),
    nonBillable: sql<number>`coalesce(sum(${fl.asanaHours}) filter (where ${fl.isBillable}=false),0)`.mapWith(Number),
  })
  .from(fl).where(and(eq(fl.orgId, orgId), eq(fl.periodKey, periodKey)));
