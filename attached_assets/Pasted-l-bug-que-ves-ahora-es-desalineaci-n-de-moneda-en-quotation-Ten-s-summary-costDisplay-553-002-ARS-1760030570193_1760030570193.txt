type Cur = 'ARS'|'USD';
const fx = { usd_ars: /* tu rate mensual */ };

function toNative(amountUSD: number, native: Cur) {
  return native === 'USD' ? amountUSD : Math.round(amountUSD * fx.usd_ars);
}

// Normalizador de quotation
function normalizeQuotation(q: any, native: Cur) {
  const totalAmount = Number(q?.totalAmount ?? 0);        // USD (legacy)
  const totalAmountNative =
    q?.totalAmountNative != null
      ? Number(q.totalAmountNative)
      : toNative(totalAmount, native); // 🔒 fallback consistente

  const estimatedHours = Number(q?.estimatedHours ?? -1);

  return { ...q, totalAmount, totalAmountNative, estimatedHours };
}

const USD_NATIVE = new Set(['Warner', 'Kimberly Clark']);
function nativeFor(client: string): 'USD'|'ARS' {
  return USD_NATIVE.has(client) ? 'USD' : 'ARS';
}

function buildProjectPayload(row: SheetRow, rates: {usd_ars:number}) {
  const cur = nativeFor(row.Cliente);
  const usd_ars = rates.usd_ars;

  const costUSD  = num(row['Costos (USD)']);
  const billUSD  = num(row['Facturación (USD)']);
  const quoteNat = num(row['Cotización']); // nativa en la hoja

  const costDisplay     = cur === 'USD' ? costUSD : Math.round(costUSD * usd_ars);
  const revenueDisplay  = cur === 'USD' ? billUSD : Math.round(billUSD * usd_ars);
  const totalAmount     = cur === 'USD' ? quoteNat : Math.round(quoteNat / usd_ars);   // opcional, legado
  const totalAmountNat  = quoteNat;                                                    // 🔒 SIEMPRE

  const markup = (costDisplay > 0 && totalAmountNat > 0) ? totalAmountNat / costDisplay : null;
  const margin = (totalAmountNat > 0) ? (totalAmountNat - costDisplay) / totalAmountNat : null;
  const budgetUtilization = (totalAmountNat > 0) ? costDisplay / totalAmountNat : null;

  return {
    summary: {
      costDisplay, revenueDisplay, currencyNative: cur, markup, margin
    },
    quotation: {
      totalAmount,                // para compatibilidad
      totalAmountNative: totalAmountNat,
      estimatedHours: row['Horas Estimadas'] ?? -1,
    },
    actuals: { totalWorkedCost: costUSD },
    metrics: { budgetUtilization }
  };
}
