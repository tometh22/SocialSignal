SQL de referencia (esqueleto)
-- Totales mes
SELECT
  project_id, month_key,
  SUM(amount_usd) AS ingresos_usd
FROM google_sheets_sales
WHERE confirmed = 'SI'
GROUP BY 1,2;

SELECT
  project_id, month_key,
  SUM(monto_total_usd) AS costos_usd,
  SUM(horas_reales_asana) AS horas_reales,
  SUM(horas_para_facturacion) AS horas_fact
FROM direct_costs
GROUP BY 1,2;

-- KPIs
SELECT
  i.project_id, i.month_key,
  i.ingresos_usd,
  c.costos_usd,
  CASE WHEN c.costos_usd>0 THEN i.ingresos_usd/c.costos_usd END AS markup,
  CASE WHEN i.ingresos_usd>0 THEN (i.ingresos_usd-c.costos_usd)/i.ingresos_usd END AS margen_pct,
  c.horas_reales, c.horas_fact,
  CASE WHEN c.horas_reales>0 THEN c.horas_fact/c.horas_reales END AS realizacion
FROM ingresos i
LEFT JOIN costos c USING(project_id, month_key);
