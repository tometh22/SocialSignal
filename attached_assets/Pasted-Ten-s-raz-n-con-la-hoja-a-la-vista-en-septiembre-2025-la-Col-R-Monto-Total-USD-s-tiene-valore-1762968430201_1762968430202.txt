Helpers robustos (evitando que “cero” y “vacío” se confundan):

const toNumber = (v:any) => {
  if (v === null || v === undefined) return null;
  if (typeof v === 'number') return v;
  const s = String(v).replace(/[^\d.-]/g,'');
  if (s === '' || s === '-' || s === '.' || s === '-.') return null;
  const n = Number(s);
  return Number.isFinite(n) ? n : null;
};

function toPeriodKey(m: string, y: number) {
  const map:any = { ene:1,feb:2,mar:3,abr:4,may:5,jun:6,jul:7,ago:8,sep:9,oct:10,nov:11,dic:12 };
  const m3 = m.slice(0,3).toLowerCase().normalize('NFD').replace(/\p{Diacritic}/gu,'');
  const mm = map[m3]; 
  return mm ? `${y}-${String(mm).padStart(2,'0')}` : null;
}
