2) Helpers críticos (parsers & fechas)
// parser.ts
export const parseNumberEs = (v: unknown): number => {
  if (v == null) return 0;
  // "4.359.857" -> "4359857", "1.750" -> "1750", "29,230.00" (por si acaso) -> "29230.00"
  const s = String(v).trim().replace(/\./g, '').replace(',', '.');
  const n = Number(s);
  return Number.isFinite(n) ? n : 0;
};

const MES = {
  'enero': 1,'febrero': 2,'marzo': 3,'abril': 4,'mayo': 5,'junio': 6,
  'julio': 7,'agosto': 8,'septiembre': 9,'setiembre': 9,'octubre':10,'noviembre':11,'diciembre':12
} as const;

export const spanishMonthToNumber = (mesEs: string): number => {
  const k = mesEs.normalize('NFD').replace(/\p{Diacritic}/gu,'').toLowerCase().trim();
  const n = (MES as any)[k];
  if (!n) throw new Error(`Mes inválido: "${mesEs}"`);
  return n;
};

export const periodKeyOf = (mesEs: string, year: number): `${number}-${string}` => {
  const m = spanishMonthToNumber(mesEs);
  return `${year}-${String(m).padStart(2,'0')}`;
};

export const isYes = (v: any) => String(v ?? '').trim().toLowerCase().replace('í','i') === 'si';

4) FX y cache
// fx.ts
const cache = new Map<string, number>(); // key: period "YYYY-MM" -> fx

export async function getFx(period: string): Promise<number> {
  if (cache.has(period)) return cache.get(period)!;
  // Lee de tu tabla "tipo de cambio" por mes; fallback explícito si no existe
  const fx = await db.fx.get(period); // p.ej. 1345 para 2025-08
  if (!fx) throw new Error(`FX no disponible para período ${period}`);
  cache.set(period, fx);
  return fx;
}

5) Reglas de negocio (filtro, normalización, agregación)
// business-rules.ts
import type { IncomeRow, ProjectIncome, IncomeResult, PortfolioIncome } from './types';
import { periodKeyOf } from './parser';
import { getFx } from './fx';

type GroupKey = string; // `${clientName}|${projectName}`

export async function buildIncomeResult(period: string, rows: IncomeRow[], totalProjects: number): Promise<IncomeResult> {
  // 1) Filtro temporal + confirmados
  const accepted = rows.filter(r => r.confirmed && periodKeyOf(r.monthEs, r.year) === period);

  // 2) Agrupar por proyecto (cliente + proyecto)
  const groups = new Map<GroupKey, IncomeRow[]>();
  for (const r of accepted) {
    if (!r.clientName || !r.projectName) continue;
    const k = `${r.clientName}|||${r.projectName}`;
    (groups.get(k) ?? groups.set(k, []).get(k)!).push(r);
  }

  const fx = await getFx(period);
  const projects: ProjectIncome[] = [];

  for (const [k, list] of groups) {
    const [clientName, projectName] = k.split('|||');

    // Regla de moneda nativa (display):
    // Si hay algún USD > 0 en el proyecto/periodo → display USD, else si hay ARS > 0 → display ARS
    const sumUSD = list.reduce((a,r)=>a+(r.amountUSD||0),0);
    const sumARS = list.reduce((a,r)=>a+(r.amountARS||0),0);

    const revenueDisplay =
      sumUSD > 0 ? { amount: sumUSD, currency: 'USD' as const } :
      sumARS > 0 ? { amount: sumARS, currency: 'ARS' as const } :
      { amount: 0, currency: 'USD' as const };

    // Normalización a USD para KPIs (regla estricta por registro)
    const revenueUSDNormalized = list.reduce((acc, r) => {
      if (r.amountUSD > 0) return acc + r.amountUSD;
      if (r.amountARS > 0) return acc + (r.amountARS / fx);
      return acc;
    }, 0);

    projects.push({
      projectId: await resolveProjectId(clientName, projectName), // o null si no aplica
      clientName, projectName,
      revenueDisplay,
      revenueUSDNormalized,
      records: list.map(r => r.amountUSD > 0
        ? ({ currency: 'USD' as const, amount: r.amountUSD })
        : ({ currency: 'ARS' as const, amount: r.amountARS })
      ),
    });
  }

  const periodRevenueUSD = projects.reduce((a,p)=>a+p.revenueUSDNormalized,0);
  const summary: PortfolioIncome = {
    period,
    periodRevenueUSD,
    projectsWithIncome: projects.filter(p => p.revenueUSDNormalized > 0).length,
    totalProjects
  };

  return { period, projects, summary };
}

// Resolver ID de proyecto (si tenés tabla de proyectos)
async function resolveProjectId(client: string, project: string): Promise<number|null> {
  const row = await db.projects.findByClientAndName(client, project);
  return row?.id ?? null;
}

6) API pública del dominio
// index.ts
import { fetchSalesRawForPeriod } from './data-access';
import { buildIncomeResult } from './business-rules';
import type { PeriodKey, IncomeResult, ProjectIncome, PortfolioIncome } from './types';

export async function getIncomeByPeriod(period: PeriodKey): Promise<IncomeResult> {
  const rows = await fetchSalesRawForPeriod(period);
  const totalProjects = await db.projects.countActive(); // o todos
  return buildIncomeResult(period, rows, totalProjects);
}

export async function getIncomeByProject(projectId: number, period: PeriodKey): Promise<ProjectIncome> {
  const { projects } = await getIncomeByPeriod(period);
  const p = projects.find(p => p.projectId === projectId);
  return p ?? {
    projectId, clientName: '', projectName: '',
    revenueDisplay: { amount: 0, currency: 'USD' },
    revenueUSDNormalized: 0, records: []
  };
}

export async function getPortfolioIncome(period: PeriodKey): Promise<PortfolioIncome> {
  const { summary } = await getIncomeByPeriod(period);
  return summary;
}
