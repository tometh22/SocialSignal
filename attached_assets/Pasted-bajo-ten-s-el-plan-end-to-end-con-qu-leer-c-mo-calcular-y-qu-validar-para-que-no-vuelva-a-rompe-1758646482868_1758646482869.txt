function parseMoney(input: unknown): number | null {
  if (typeof input === 'number') return input;
  if (typeof input !== 'string') return null;

  // quitar currency & espacios
  let s = input.replace(/[^\d.,-]/g, '').trim();

  // decidir decimal (último separador) y remover el resto (miles)
  const lastComma = s.lastIndexOf(',');
  const lastDot   = s.lastIndexOf('.');
  const dec = Math.max(lastComma, lastDot);
  if (dec !== -1) {
    const decimal = s[dec];                    // ',' o '.'
    s = s.replace(/[.,]/g, '');                // quito todos
    s = s.slice(0, dec) + '.' + s.slice(dec);  // re-inserto punto decimal
  }
  const n = Number(s);
  if (!isFinite(n)) return null;

  // 🛡️ anti ×100: si n>1e6 y existe una versión n/100 coherente, elegir la razonable
  if (n > 1_000_000) {
    const n2 = n / 100;
    // heurísticas: n2 ∈ (1..500_000) y se parece a medianas de ese cliente/proyecto
    return n2;
  }
  return n;
}
Esto corrige automáticamente Warner (2,923,000 → 29,230).
