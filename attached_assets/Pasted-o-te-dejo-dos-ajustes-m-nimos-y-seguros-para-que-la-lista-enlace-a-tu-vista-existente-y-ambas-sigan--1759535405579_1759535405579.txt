3) Heads-up: hay un bugcito de sintaxis en tu archivo

En el código que pegaste, periodToLabel quedó con llaves duplicadas y te va a romper el build. Reemplazalo por este:

function periodToLabel(period: string): string {
  try {
    const [y, m] = period.split("-").map(Number);
    const d = new Date(y, m - 1, 1);
    return d.toLocaleDateString("es", { month: "short", year: "numeric" });
  } catch {
    return period;
  }
}
