1.2 Generar monthKey robusto
const M = {
  'enero':'01','febrero':'02','marzo':'03','abril':'04','mayo':'05','junio':'06',
  'julio':'07','agosto':'08','septiembre':'09','octubre':'10','noviembre':'11','diciembre':'12',
  'ene':'01','feb':'02','mar':'03','abr':'04','may':'05','jun':'06','jul':'07','ago':'08','sep':'09','oct':'10','nov':'11','dic':'12'
};
function monthKeyFrom(mesTxt: string, anio: number) {
  const s = mesTxt.toLowerCase().trim();
  // casos “08 ago”, “agosto”, “ago”
  const mm = (s.match(/\b(\d{2})\b/)?.[1]) ?? M[s.split(/\s+/)[0]]; 
  if (!mm) throw new Error(`Mes no reconocido: ${mesTxt}`);
  return `${anio}-${mm}`;
}

1.3 Parseo de montos/horas seguro
function parseMoney(x: any){ 
  if (x == null || x === '') return 0;
  const n = Number(String(x).replace(/[^\d.-]/g,'').replace(',', '.'));
  // Anti ×100: si huele a export mal escalada
  return (n > 1_000_000 && n % 100 === 0) ? n/100 : n;
}
function parseHours(x:any){
  if (x == null || x === '') return 0;
  return Number(String(x).replace(',','.'));
}
