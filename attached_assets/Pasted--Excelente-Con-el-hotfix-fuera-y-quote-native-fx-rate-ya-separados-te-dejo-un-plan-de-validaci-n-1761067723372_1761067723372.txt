3) Validación SQL (BD → verdad única)

3.1. Totales Labor por período (invariantes)
-- Labor: las sumas deben ser consistentes
SELECT
  period_key,
  ROUND(SUM(target_hours)::numeric,2)  AS target_h,
  ROUND(SUM(asana_hours)::numeric,2)   AS asana_h,
  ROUND(SUM(billing_hours)::numeric,2) AS billing_h,
  ROUND(SUM(cost_ars)::numeric,2)      AS total_ars,
  ROUND(SUM(cost_usd)::numeric,2)      AS total_usd
FROM fact_labor_month
WHERE project_id = 39 AND period_key IN ('2025-08','2025-09')
GROUP BY period_key
ORDER BY period_key;
3.2. RC por período (ingresos / cotización / FX)
SELECT
  period_key,
  CAST(revenue_usd AS numeric(12,2))   AS revenue_usd,
  CAST(revenue_ars AS numeric(14,2))   AS revenue_ars,
  CAST(cost_usd    AS numeric(12,2))   AS cost_usd,
  CAST(cost_ars    AS numeric(14,2))   AS cost_ars,
  CAST(quote_native AS numeric(12,2))  AS quote_native, -- precio proyecto (denominador)
  CAST(fx_rate      AS numeric(10,4))  AS fx_rate        -- tipo de cambio
FROM fact_rc_month
WHERE project_id = 39 AND period_key IN ('2025-08','2025-09')
ORDER BY period_key;

quote_native ≠ 0 / NULL (p.ej. 8450 para KC Ago).
fx_rate correcto (p.ej. 1345).
cost_usd / cost_ars compatibles con FX y moneda del período.
3.3. Cruce Labor vs RC (no deben “pegar saltos”)
-- Diferencia absoluta entre costos RC y suma Labor (en USD, si existe)
WITH l AS (
  SELECT period_key, SUM(cost_usd) AS labor_usd
  FROM fact_labor_month
  WHERE project_id=39 AND period_key IN ('2025-08','2025-09')
  GROUP BY period_key
),
r AS (
  SELECT period_key, cost_usd
  FROM fact_rc_month
  WHERE project_id=39 AND period_key IN ('2025-08','2025-09')
)
SELECT
  COALESCE(l.period_key,r.period_key) AS period_key,
  ROUND(l.labor_usd::numeric,2) AS labor_usd,
  ROUND(r.cost_usd::numeric,2)  AS rc_usd,
  ROUND((COALESCE(r.cost_usd,0)-COALESCE(l.labor_usd,0))::numeric,2) AS delta_usd
FROM l FULL JOIN r USING (period_key)
ORDER BY period_key;
OK: delta_usd ~ 0 (tolerancia chica por redondeo).
