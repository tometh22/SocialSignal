1) Unificar fuente: selector único de “costDisplay”
Crea un selector y úsalo en lista y detalle. Prohíbe conversiones en el render.
// selectors/projectVM.ts
type Currency = 'ARS' | 'USD';
type ProjectState = {
  summary?: { costDisplay?: number; revenueDisplay?: number; currencyNative?: Currency };
  actuals?: { totalWorkedCost?: number };        // USD "base"
  fx?: { usd_ars?: number };                     // opcional si querés fallback
};

const round2 = (n: number) => Math.round((n + Number.EPSILON) * 100) / 100;

export function toProjectVM(state: ProjectState) {
  const currency = state.summary?.currencyNative ?? 'ARS';

  // **Regla de oro**: si viene summary.costDisplay, es FINAL (no convertir)
  let costDisplay =
    state.summary?.costDisplay ??
    (currency === 'ARS' && state.fx?.usd_ars && state.actuals?.totalWorkedCost != null
      ? round2(state.actuals.totalWorkedCost * state.fx.usd_ars)  // fallback consciente
      : state.actuals?.totalWorkedCost ?? 0);

  return {
    currencyNative: currency,
    costDisplay,
    revenueDisplay: state.summary?.revenueDisplay ?? 0
  };
}

// en el componente padre de detalle (solo DEV)
useEffect(() => {
  if (process.env.NODE_ENV === 'production') return;
  const s = state.summary?.costDisplay;
  const a = state.actuals?.totalWorkedCost;
  if (s && a) {
    const ratio = s / a;
    // si ratio parece un tipo de cambio, alguien está mostrando USD en vez de summary

      console.error('[CONSISTENCY] Detalle usando USD en UI. Esperado summary.costDisplay en moneda nativa.', { summaryCostDisplay: s, actualsUSD: a, ratio });

  }
}, [state.summary?.costDisplay, state.actuals?.totalWorkedCost]);
3) Hook “¿de dónde sale este número?” (traza inmediata)
Para saber QUÉ card está ignorando summary, pegá este helper y llámalo donde pintás montos:
function useWhichCost(valueShown: number, state: ProjectState, tag: string) {
  useEffect(() => {
    const s = state.summary?.costDisplay;
    const a = state.actuals?.totalWorkedCost;
    const cur = state.summary?.currencyNative;
    if (s != null && valueShown === s) {
      console.log(`[COST OK][${tag}] summary.costDisplay (${cur}) =`, valueShown);
    } else if (a != null && valueShown === a) {
      console.warn(`[COST MISMATCH][${tag}] mostrando actuals.totalWorkedCost (USD) =`, valueShown);
    } else {
      console.log(`[COST][${tag}] valor derivado =`, valueShown, { cur, s, a });
    }
  }, [valueShown, state.summary?.costDisplay, state.actuals?.totalWorkedCost, state.summary?.currencyNative, tag]);
}
