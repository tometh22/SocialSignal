// routes/deviation.ts (ESM)
import type { Request, Response } from 'express';
import { resolveTimeFilter } from '../services/time.js';
import { computeProjectPeriodMetrics } from '../domain/metrics.js';

const sev = (pct: number) => {
  const ap = Math.abs(pct);
  if (ap >= 30) return 'critical';
  if (ap >= 15) return 'high';
  if (ap >= 5)  return 'medium';
  return 'low';
};

export async function deviationAnalysisHandler(req: Request, res: Response) {
  try {
    const projectId = String(req.params.id ?? req.query.projectId ?? '');
    const timeFilter = String(req.query.timeFilter ?? '');
    const basis = (String(req.query.basis ?? 'ECON').toUpperCase() === 'EXEC') ? 'EXEC' : 'ECON';

    if (!projectId || !timeFilter) {
      return res.status(400).json({ error: 'projectId and timeFilter are required' });
    }

    const tf = resolveTimeFilter(timeFilter); // {start,end, period:'YYYY-MM'}
    const data = await computeProjectPeriodMetrics(projectId, tf, basis);

    // data.teamBreakdown debe traer por persona: K, L, M, budgetUSD, actualUSD, name, id
    const deviations = (data.teamBreakdown ?? []).map(p => {
      const K = Number(p.K ?? 0);
      const L = Number(p.L ?? 0);
      const deviationPct = K === 0 ? 0 : ((L / K) - 1) * 100;
      return {
        personnelId: p.personId ?? p.id ?? p.name,
        personnelName: p.name,
        budgetedHours: K,
        actualHours: L,
        budgetedCost: Number(p.budgetUSD ?? 0),
        actualCost: Number(p.actualUSD ?? 0),
        hourDeviation: L - K,
        costDeviation: Number(p.actualUSD ?? 0) - Number(p.budgetUSD ?? 0),
        deviationPercentage: deviationPct,
        severity: sev(deviationPct),
        alertType: K === 0 ? 'no-plan' : (L >= K ? 'overrun' : 'underrun'),
        deviationType: 'hours',
      };
    });

    return res.json({
      summary: {
        period: data.summary?.period,      // 'YYYY-MM'
        basis,
        activeMembers: (data.teamBreakdown ?? []).filter(p => (p.L ?? 0) > 0).length,
        totalHours: data.summary?.totalHours,
        efficiencyPct: data.summary?.efficiencyPct,
        teamCost: basis === 'ECON' ? data.summary?.teamCostUSD : data.summary?.teamCostUSD, // mismo nombre en front
        emptyStates: data.summary?.emptyStates,
        hasData: data.summary?.hasData,
      },
      deviations
    });
  } catch (e:any) {
    return res.status(500).json({ error: 'deviation-analysis failed', detail: e?.message });
  }
}
